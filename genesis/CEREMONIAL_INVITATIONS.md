## Implementation Notes

### QR Code Generation
```typescript
import QRCode from 'qrcode';

const generateQR = async (url: string, outputPath: string) => {
  await QRCode.toFile(outputPath, url, {
    errorCorrectionLevel: 'H',
    type: 'png',
    quality: 1,
    margin: 2,
    color: {
      dark: '#000000',
      light: '#F59E0B' // Amber background
    },
    width: 512
  });
};
```
